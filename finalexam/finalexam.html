Planning Document

Scope : To develop a Java Animation Applet that simulates 2 elevators in a 5 storey building.

Main Enhancements :

  1. Lights or numbers in a display showing what floor each elevator is on.

  2. Animation of elevator doors opening and closing.

  3. A graphic image of the interior of the elevator.

  4. An animation of the elevator doors opening when they get to a floor and closing when a button is pressed to go to another floor.

  5. An animated table telling which floors are waiting for an elevator.

  6. The ability to have more than one button active in an elevator, so that it can stop on multiple floors on one trip.

  7. An animator table showing where each elevator is going.

  8. Adding sound effects.

Primary Actor: User who loads the page to view the elevator applet.

Stakeholders and Interests:

-Person who loads the web page

      Is interested to successfully load and use the elevator Applet

Pre-conditions:

Internet connection is established.

The web page is loaded successfully.

The applet is loaded successfully.

Success Guarantee - Post Conditions:

The elevator Applet is loaded successfully and the user is able to view and operate the different functionalities of the elevator.

Main Success Scenario (or Basic Flow):

1. User clicks on the link directing to the elevator Applet.

2. The web page with the applet opens.

3. The contents of the applet are displayed.

Extensions or Alternative Flows:

1. At any time, the internet connection is lost.

    a. Error message displayed requesting the user to check his internet connection.

2. The user clicks the back button before the program is loaded fully.

    b. The program terminates and the previous web page is displayed.

package Elevator;

 

import java.awt.Color;

import java.awt.Font;

import java.awt.Graphics;

import java.awt.Rectangle;

 

import javax.swing.JPanel;

 

@SuppressWarnings("serial")

public class Building extends JPanel implements Draw

{

  /** Static number of FloorCount in the building */

  public static final int FloorCount = 5;

 

  /** Elevator #1 */

  protected Elevator Elevator1;

 

  /** Elevator #2 */

  protected Elevator Elevator2;

 

  /**

   * Building constructor

   */

  public Building()

  {

    // Create the elevators

    Elevator1 = new Elevator();

    Elevator2 = new Elevator();

  }

 

  /**

   * Get elevator #1

   *

   * @return Elevator 1 reference

   */

  public Elevator getElevator1()

  {

    return Elevator1;

  }

 

  /**

   * Get elevator #2

   *

   * @return Elevator 2 reference

   */

  public Elevator getElevator2()

  {

    return Elevator2;

  }

 

  @Override

  public void draw(Graphics g, Rectangle bounds)

  {

        

    int width  = (int)(bounds.width * 0.9);

    int height = (int)(bounds.height * 0.9);

    int x = bounds.width/2 - width/2;

    int y = bounds.height - height -1;

   

    // Draw the building

    g.setColor(Color.CYAN);

    g.fillRect(x, y, width/2, height);

   

 //   ((Graphics2D)g).setStroke(new BasicStroke(2.0f));

    g.setColor(Color.MAGENTA);

    g.drawRect(x, y, width/2, height);

   

    // Draw each floor line

    int nextFloor = y;

    String S ;

    Font f = new Font(Font.SANS_SERIF, 4, 20);   

    for(int i = 0; i < FloorCount; i++)

    {

      S = "" + (FloorCount - i);

      nextFloor += (height/FloorCount);

      g.drawLine(x, nextFloor, x+width/2, nextFloor);

      g.setFont(f);

      g.drawString( S, x + 10, nextFloor -3);

    }

   

    // Draw the building

    g.setColor(Color.yellow);

    g.fillRect(x + width /2, y, width/2, height);

   

 //   ((Graphics2D)g).setStroke(new BasicStroke(2.0f));

    g.setColor(Color.red);

    g.drawRect(x + width /2, y, width/2, height);

 

 

     nextFloor = y;

    for(int i = 0; i < FloorCount; i++)

    {

      S = "" + (FloorCount - i);

      nextFloor += (height/FloorCount);

      g.drawLine(x+width/2, nextFloor, x+width, nextFloor);

      g.setFont(f);

      g.drawString( S, x+width/2 + 10, nextFloor - 3 );

    }   

   

  // ((Graphics2D)g).setStroke(new BasicStroke(1.0f));

 

    // Draw the elevators

    int centerLine     = width/2;

    int elevatorWidth  = width/4;

    int elevatorHeight = height - 6;

    int elevatorY      = y + 3;

   

    // Draw elevator #1

    int e1X = centerLine/2 - elevatorWidth/2 + x;

    Rectangle e1Bounds = new Rectangle(e1X, elevatorY, elevatorWidth, elevatorHeight);

    Elevator1.draw(g, e1Bounds);

   

    // Draw elevator #2

    int e2X = (int)(width * 0.75) - elevatorWidth/2 + x;

    Rectangle e2Bounds = new Rectangle(e2X, elevatorY, elevatorWidth, elevatorHeight);

    Elevator2.draw(g, e2Bounds);

  }

}

package Elevator;

 

import java.awt.Color;

import java.awt.Dimension;

import java.awt.Graphics;

import java.awt.Rectangle;

import java.util.ArrayList;

import java.util.List;

 

import javax.swing.JPanel;

 

@SuppressWarnings("serial")

public class Canvas extends JPanel

{

  /** List of all object that can be drawn */

  List<Draw> drawables;

 

  /**

   * DrawingCanvas constructor

   */

  public Canvas()

  {

    // Enable double-buffering

    super(true);

   

    drawables = new ArrayList<Draw>();

  }

 

  /**

   * Add a drawable item to the collection

   *

   * @param d Drawable reference

   */

  public void addDrawable(Draw d)

  {

    if(!drawables.contains(d))

    {

      drawables.add(d);

    }

  }

 

  /**

   * Paint the canvas and its contents

   *

   * @param g Graphics context

   */

  @Override

  public void paint(Graphics g)

  {

    Dimension size = getSize();

    Rectangle bounds = new Rectangle(size);

   

    // Clear the background

    g.setColor(Color.WHITE);

    g.fillRect(0, 0, size.width, size.height);

   

    // Draw all drawables

    for(Draw d : drawables)

    {

      d.draw(g, bounds);

 

    }

  }

}

package Elevator;

 

import java.awt.Color;

import java.awt.Graphics;

import java.awt.Rectangle;

 

public class MovingElevator implements Draw

{

  /** Direction of travel constants */

  public static final int UP   = -1;

  public static final int DOWN = 1;

 

  /** The elevator to which this car belongs */

  protected Elevator elevator;

 

  /** Array containing the y position of each floor */

  protected int[] floorPositions = new int[Building.FloorCount];

 

  /** Current floor we are/were at */

  protected int currentFloor;

 

  /** Next floor destination */

  protected int destination;

 

  /** Current direction of travel */

  protected int direction = UP;

 

  /** The current y position of the car */

  protected int currentPosition;

 

  /** Indicates if car is currently in motion */

  protected boolean bMoving;

 

  // Various variables for tracking door state (should probably be its own class)

  boolean bDoorsClosed  = true;

  boolean bDoorsClosing = false;

  boolean bDoorsOpening = false;

  int doorWidth = 0;

  int openedAmount = 0;

 

  /** Needed for one-time initialization of the car */

  boolean bInitialized = false;

 

 

  /**

   * Car constructor

   *

   * @param floor Starting floor

   * @param elevator The owning elevator

   */

  public MovingElevator(int floor, Elevator elevator)

  {

    this.currentFloor = floor;

    this.elevator = elevator;

    elevator.notifyArrived(floor);

  }

 

  /**

   * Set the destination floor for this car

   *

   * @param destinationFloor Destination floor

   */

  public void setDestination(int destinationFloor)

  {

    if(destinationFloor != currentFloor)

    {

      closeDoors();

      destination = destinationFloor;

      direction = destination > currentFloor ? UP : DOWN;

      bMoving = true;

    }

  }

 

  /**

   * Sets indication that car has reached its destination

   */

  protected void setArrived()

  {

    bMoving = false;

    currentFloor = destination;

    elevator.notifyArrived(currentFloor);

   

    openDoors();

  }

 

  /**

   * Opens the elevator doors

   */

  protected void openDoors()

  {

    if(bDoorsClosed)

    {

      openedAmount = 0;

      bDoorsOpening = true;

      bDoorsClosing = false;

      bDoorsClosed  = false;

    }

  }

 

  /**

   * Closes the elevator doors

   */

  protected void closeDoors()

  {

    if(!bDoorsClosed)

    {

      openedAmount = doorWidth;

      bDoorsClosing = true;

      bDoorsOpening = false;

    }

  }

 

  @Override

  public void draw(Graphics g, Rectangle bounds)

  {

    // These are constant

    int w = bounds.width - 4;

    int h = bounds.height / Building.FloorCount;

    int x = bounds.x + 2;

    doorWidth = (int)(w * 0.75);

   

    // Check if we need to do one-time initialization

    if(!bInitialized)

    {

      // Determine positions for each floor

      for(int i = 0; i < Building.FloorCount; i++)

      {

        floorPositions[i] = bounds.y + bounds.height - (h * (i+1));

      }

     

      // Set current position to starting floor

      currentPosition = floorPositions[currentFloor-1];

      bInitialized = true;

    }

    else if(bMoving && bDoorsClosed) // Only move elevator if doors are closed

    {

      // Elevator is moving so calculate new position

      currentPosition += (1 * direction);

 

      // Check if we've arrived at our destination

      if(currentPosition == floorPositions[destination-1])

      {

      

 

        setArrived();

      }

      else

      {

        // Check if we've passed a floor along the way

        for(int i = 0; i < floorPositions.length; i++)

        {

          if(currentPosition == floorPositions[i])

          {

  

            elevator.notifyArrived(i+1);

            break;

          }

        }

      }

    }

   

 

 

      

    // Draw the elevator

    g.setColor(Color.ORANGE);

    g.fillRect(x, currentPosition, w, h);

 

    // Draw the doors (if needed)

    if(bDoorsOpening)

    {

      if(openedAmount < doorWidth)

      {

        openedAmount += 2;

      }

     

      g.setColor(Color.WHITE);

      int center = x + w/2;

      g.fillRect(center - openedAmount/2, currentPosition, openedAmount, h);

    }

    else if(bDoorsClosing)

    {

      openedAmount -= 2;

 

      if(openedAmount > 0)

      { 

        g.setColor(Color.WHITE);

        int center = x + w/2;

        g.fillRect(center - openedAmount/2, currentPosition, openedAmount, h);

      }

      else

      {

        // Doors are now closed

        bDoorsClosed = true;

      }

    }

  }

}  

ELEVATOR APPLET

package Elevator;

 

import java.applet.AudioClip;

import java.awt.Color;

import java.awt.Graphics;

import java.awt.Rectangle;

import java.util.HashSet;

import java.util.Set;

 

public class Elevator

{

  /** The elevator car */

  protected MovingElevator car;

 

  /** Listeners of elevator events */

  protected Set<ActListener> listeners;

 

  /** Bell sound file */

  protected AudioClip bell;

 

  /**

   * Elevator constructor

   */

  public Elevator()

  {

    listeners = new HashSet<ActListener>();

   

    // Cars start on the 1st floor

    car = new MovingElevator(1, this);

  }

 

  /**

   * Call the elevator car to the given floor

   *

   * @param floor Destination floor

   */

  public void call(int floor)

  {

    car.setDestination(floor);

  }

 

  /**

   * Add an elevator listener

   *

   * @param listener ElevatorListener reference

   */

  public void addListener(ActListener listener)

  {

    listeners.add(listener);

  }

 

  /**

   * Notify listeners when car has reached its destination floor

   *

   * @param floor The destination floor

   */

  public void notifyArrived(int floor)

  {

      

    if(bell != null)

    {

      bell.stop();

      bell.play();

    }

   

    for(ActListener listener : listeners)

    {

      listener.setCurrentFloor(this, floor);

    }

  }

 

  /**

   * Set the bell sound to be played when

   * elevator reaches each floor

   *

   * @param bell Bell audio clip

   */

  public void setBellSound(AudioClip bell)

  {

    this.bell = bell;

  }

 

 

  public void draw(Graphics g, Rectangle bounds)

  {

    int x = bounds.x;

    int y = bounds.y;

    int w = bounds.width;

    int h = bounds.height;

 

    // Clear the background

    g.setColor(Color.WHITE);

    g.fillRect(x, y, w, h);

   

    // Draw the elevator shaft

    g.setColor(Color.BLACK);

    g.drawRect(x, y, w, h);

   

    // Draw the elevator car

    car.draw(g, bounds);

  }

}
